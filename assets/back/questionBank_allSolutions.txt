def mathOperations(operation,int1,int2):
	if(operation=='-'):
		return(int1-int2)
	elif(operation=='+'):
		return(int1+int2)
	elif(operation=='*'):
		return(int1*int2)
	elif(operation=='/'):
		return(int1/int2)
	elif(operation=='%'):
		return(int1%int2)
	else:
		return 0

def fibonacci(num):
	if(num==0):
		return '0'
	elif(num==1):
		return '0'
	elif(num==2):
		return '0,1'
	else:
		nums=['0','1']
		for i in range(num-2):
			nextNum=int(nums[i])+int(nums[i+1])
			nums.append(str(nextNum))
	return ','.join(nums)

def listOverlap(listOfNumsA,listOfNumsB):
	overlap=[]
	for num in listOfNumsA:
		if(num in listOfNumsB and num not in overlap):
			overlap.append(num)
	for num in listOfNumsB:
		if(num in listOfNumsA and num not in overlap):
			overlap.append(num)
	return overlap

def caesarEncrypt(text):
	alpha='abcdefghijklmnopqrstuvwxyz'
	APLPHA=alpha.upper()
	encryptedText=''
	for char in text:
		if(char.isupper()):
			index=ALPHA.find(char)
			encryptedText=encryptedText+ALPHA[index+3]
		else:
			index=alpha.find(char)
			encryptedText=encryptedText+alpha[index+3]
	return encryptedText

def isAMinHeap(A):
	j=len(A)-1
	if((j+1)==1 or (j+1)==0):
		return True
	k=int((j-1)/2)
	if(A[j]>A[k]):
		return(isAMinHeap(A[0:j]))
	else:
		return False

def allOdds(n):
	i=0
	nums=[]
	while(i<=n):
		if(i % 2 != 0):
			nums.append(i)
		i=i+1
	return nums

def factorial(num):
	val=1
	for i in range(1,num+1):
		val=val*i
	return val
